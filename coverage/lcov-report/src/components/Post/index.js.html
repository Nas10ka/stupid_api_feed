import React, { Component } from 'react';
import moment from 'moment';
import PropTypes from 'prop-types';
import Like from '../Like';
import Styles from './styles';
 
export default class Post extends Component {
    constructor () {
        super();
        this.deletePost = ::this.deletePost;
    }
    static contextTypes = {
        options: PropTypes.object.isRequired
    };
 
    deletePost () {
        this.props.deletePost(this.props.id);
    }
 
    componentWillMount () {
        // console.log(this.props.id, 'component will Mount');
    }
    componentDidMount () {
        // console.log(this.props.id, 'component did Mount');
    }
    // shouldComponentUpdate () {
    //     return false;
    // }
 
    render () {
        const {
            id,
            likes,
            likePost,
            avatar,
            comment,
            firstName,
            lastName,
            created,
            ownfirstName,
            ownlastName
        } = this.props;
        // console.log(`${firstName} ${lastName}`);
 
        return (
            <div className = { Styles.post }>
                {`${ownfirstName} ${ownlastName}` ===
                `${firstName} ${lastName}` ? (
                    <span onClick = { this.deletePost } className = { Styles.close }>
                            {' '}
                        &times;
                        </span>
                    ) : null}
                <div>
                    <img alt = { firstName } src = { avatar } />
                </div>
                <a href = '#'>{`${firstName} ${lastName}`}</a>
                <time>{moment.unix(created).format('MMMM D h:mm:ss a')}</time>
 
                {comment.substring(0, 4) === 'http' ? (
                    <div className = { Styles.img }>
                        <img src = { comment } alt = 'img' />
                    </div>
                ) : (
                    <p>{comment}</p>
                )}
                {/*<p>{comment}</p>*/}
 
                <Like id = { id } likes = { likes } likePost = { likePost } />
            </div>
        );
    }
}
